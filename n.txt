    <script>
        function downloadRoutine() {
            const element = document.getElementById('routine-sheet');
            const button = document.querySelector('.btn-download');
            
            button.innerHTML = "PROCESSING...";
            button.style.opacity = "0.7";
            
            setTimeout(() => {
                html2canvas(element, {
                    scale: 4, // Ultra High Quality
                    useCORS: true,
                    backgroundColor: "#ffffff",
                    windowWidth: 1400 // Force desktop width for the image
                }).then(canvas => {
                    const link = document.createElement('a');
                    link.download = 'Routine-Monochrome.png';
                    link.href = canvas.toDataURL('image/png');
                    link.click();
                    
                    button.innerHTML = "DOWNLOAD HIGH-RES PNG";
                    button.style.opacity = "1";
                });
            }, 100);
        }
    </script>
